docker run <id> - запуск контейнера (сначала скачает если потребуется)
docker start - запуск контецнера
docker stop - остановка контейнера
docker ps - список всех запущенных контейнеров

docker ps -a - история всех запущенных контейнеров и остановленных

docker run -p6000:6729 redis -смена привзки порта
docker run --name и названаие контейнера
